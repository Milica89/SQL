-- predmeti koje nije polozio nijedan student

SELECT * FROM PREDMET P
WHERE NOT EXISTS (SELECT * FROM ISPIT I WHERE P.ID=I.PREDMET_ID)

-- nastavnici koji predaju vise od jednog predmeta

SELECT N.IME, N.PREZIME FROM NASTAVNIK N, PREDAJE PR
WHERE N.ID=PR.NASTAVNIK_ID
GROUP BY N.IME, N.PREZIME HAVING COUNT(PR.PREDMET_ID)>1

-- predmeti koje je polozila Tamara Novakovic

SELECT P.NAZIV FROM PREDMET P
WHERE P.ID IN (SELECT I.PREDMET_ID FROM ISPIT I, STUDENT S 
WHERE S.IME='Tamara' AND S.PREZIME='Novakovic' AND S.ID=I.STUDENT_ID)

-- broj polozenih predmeta za svakog studenta

SELECT S.IME, S.PREZIME, COUNT(I.PREDMET_ID) FROM STUDENT S, ISPIT I
WHERE S.ID=I.STUDENT_ID
GROUP BY S.IME, S.PREZIME
