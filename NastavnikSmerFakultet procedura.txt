--za datog nastavnika vraca na kom smeru i fakultetu radi

CREATE PROCEDURE SmerFakultetZaNastavnika (@nastavnikId int)
AS
BEGIN
	SELECT S.NAZIV +', '+ F.NAZIV 
	FROM NASTAVNIK N
	JOIN PREDAJE PR ON PR.NASTAVNIK_ID=N.ID
	JOIN PREDMET P ON P.ID=PR.PREDMET_ID
	JOIN PREDMET_SMER_GODINA PSG ON P.ID=PSG.PREDMET_ID
	JOIN SMER S ON S.ID=PSG.SMER_ID
	JOIN FAKULTET F ON F.ID=S.FAKULTET_ID
	WHERE N.ID=@nastavnikId
	END
GO
